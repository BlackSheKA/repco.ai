```sql
-- 00029_identity_split.sql
-- Phase 17.8: split social_accounts.handle into display_name + platform-specific
-- identity columns. All current users are test data (memory rule); TRUNCATE is safe.
-- Implements CONTEXT.md D-01, D-02, D-03, D-08, D-11.

BEGIN;

-- 1. New ENUM for identity resolution lifecycle (D-08).
CREATE TYPE identity_status AS ENUM ('pending', 'resolved', 'failed');

-- 2. Wipe rows. Phase 17.5-01 already TRUNCATE'd on its migration; dev/prod
-- accounts already need reconnection. CASCADE drops downstream FK references.
TRUNCATE TABLE social_accounts CASCADE;

-- 3. Drop the conflated column.
ALTER TABLE social_accounts DROP COLUMN handle;

-- 4. Add the four new columns.
ALTER TABLE social_accounts
  ADD COLUMN display_name TEXT NOT NULL,
  ADD COLUMN reddit_username TEXT,
  ADD COLUMN linkedin_slug TEXT,
  ADD COLUMN identity_status identity_status NOT NULL DEFAULT 'pending';

-- 5. CHECK: identity column matches platform (D-02). NULL allowed while pending.
ALTER TABLE social_accounts ADD CONSTRAINT social_accounts_identity_matches_platform
  CHECK (
    (platform = 'reddit'   AND linkedin_slug   IS NULL) OR
    (platform = 'linkedin' AND reddit_username IS NULL)
  );

-- 6. Partial UNIQUE indexes (D-03): one identity per user per platform; NULLs ok.
CREATE UNIQUE INDEX social_accounts_user_reddit_uidx
  ON social_accounts (user_id, reddit_username)
  WHERE reddit_username IS NOT NULL;

CREATE UNIQUE INDEX social_accounts_user_linkedin_uidx
  ON social_accounts (user_id, linkedin_slug)
  WHERE linkedin_slug IS NOT NULL;

-- 7. Hot-path index for worker gating (.eq('identity_status','resolved')).
CREATE INDEX social_accounts_user_resolved_idx
  ON social_accounts (user_id, platform)
  WHERE identity_status = 'resolved';

COMMIT;
```
