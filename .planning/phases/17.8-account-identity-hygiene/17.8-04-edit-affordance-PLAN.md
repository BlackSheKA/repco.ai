```ts
"use server"

// ... existing imports
import { resolveAccountIdentity } from "@/features/accounts/server/resolve-account-identity"

const DISPLAY_NAME_MAX = 64

export async function updateDisplayName(
  accountId: string,
  rawDisplayName: string,
): Promise<{ success: true } | { error: "unauthorized" | "empty" | "too_long" | "not_found" | "db_error" }> {
  const supabase = await createClient() // existing user-scoped client (RLS will gate)
  const { data: userResult } = await supabase.auth.getUser()
  if (!userResult?.user) return { error: "unauthorized" }

  const trimmed = rawDisplayName.trim()
  if (trimmed.length === 0) return { error: "empty" }
  if (trimmed.length > DISPLAY_NAME_MAX) return { error: "too_long" }

  const { error, data } = await supabase
    .from("social_accounts")
    .update({ display_name: trimmed })
    .eq("id", accountId)
    .eq("user_id", userResult.user.id) // belt-and-suspenders; RLS already enforces
    .select("id")
    .maybeSingle()

  if (error) {
    console.warn("[17.8] updateDisplayName db_error", { accountId, error })
    return { error: "db_error" }
  }
  if (!data) return { error: "not_found" }

  revalidatePath("/accounts")
  return { success: true }
}

export async function retryAccountIdentity(
  accountId: string,
): Promise<{ success: true } | { error: "unauthorized" | "not_found" | "session_unavailable" | "extract_failed" | "validation_failed" | "duplicate" | "db_error" }> {
  const supabase = await createClient()
  const { data: userResult } = await supabase.auth.getUser()
  if (!userResult?.user) return { error: "unauthorized" }

  // Confirm ownership (RLS would also catch this, but explicit guard is cheap).
  const { data: account } = await supabase
    .from("social_accounts")
    .select("id")
    .eq("id", accountId)
    .eq("user_id", userResult.user.id)
    .maybeSingle()
  if (!account) return { error: "not_found" }

  return resolveAccountIdentity(accountId, { forceFreshSession: true })
}
```

**A. Create `src/features/accounts/components/edit-display-name-popover.tsx`** (`"use client"`):

```tsx
"use client"

import { useState, useTransition } from "react"
import { Pencil, Loader2 } from "lucide-react"
import { toast } from "sonner"
import { Popover, PopoverContent, PopoverTrigger } from "@/components/ui/popover"
import { Button } from "@/components/ui/button"
import { Input } from "@/components/ui/input"
import { updateDisplayName } from "@/features/accounts/actions/account-actions"

const MAX = 64

export function EditDisplayNamePopover({
  accountId,
  currentDisplayName,
}: {
  accountId: string
  currentDisplayName: string
}) {
  const [open, setOpen] = useState(false)
  const [value, setValue] = useState(currentDisplayName)
  const [inlineError, setInlineError] = useState<string | null>(null)
  const [isPending, startTransition] = useTransition()

  const trimmed = value.trim()
  const disabled = trimmed.length === 0 || trimmed === currentDisplayName || isPending

  function onSubmit(e: React.FormEvent) {
    e.preventDefault()
    if (disabled) return
    setInlineError(null)
    startTransition(async () => {
      const result = await updateDisplayName(accountId, value)
      if ("success" in result) {
        toast.success("Display name updated")
        setOpen(false)
      } else if (result.error === "empty") {
        setInlineError("Name can't be empty")
      } else if (result.error === "too_long") {
        setInlineError("Keep it under 64 characters")
      } else {
        toast.error("Couldn't save the name — try again")
      }
    })
  }

  return (
    <Popover open={open} onOpenChange={setOpen}>
      <PopoverTrigger asChild>
        <Button
          variant="ghost"
          size="icon"
          className="h-7 w-7 text-muted-foreground"
          aria-label={`Edit display name for ${currentDisplayName}`}
        >
          <Pencil className="h-3.5 w-3.5" />
        </Button>
      </PopoverTrigger>
      <PopoverContent className="w-72" align="start">
        <form onSubmit={onSubmit} className="flex flex-col gap-2">
          <label htmlFor={`display-name-${accountId}`} className="text-sm font-medium">
            Display name
          </label>
          <Input
            id={`display-name-${accountId}`}
            value={value}
            onChange={(e) => setValue(e.target.value)}
            maxLength={MAX}
            autoFocus
          />
          <p className="text-xs text-muted-foreground">{trimmed.length}/{MAX} characters</p>
          {inlineError && <p className="text-sm text-destructive">{inlineError}</p>}
          <div className="flex justify-end gap-2 pt-2">
            <Button type="button" variant="ghost" size="sm" onClick={() => setOpen(false)}>
              Cancel
            </Button>
            <Button type="submit" size="sm" disabled={disabled}>
              {isPending ? (
                <>
                  <Loader2 className="mr-2 h-4 w-4 animate-spin" />
                  Saving…
                </>
              ) : (
                "Save"
              )}
            </Button>
          </div>
        </form>
      </PopoverContent>
    </Popover>
  )
}
```

**B. Create `src/features/accounts/components/retry-identity-button.tsx`** (`"use client"`):

```tsx
"use client"

import { useTransition } from "react"
import { RefreshCw, Loader2 } from "lucide-react"
import { toast } from "sonner"
import { Button } from "@/components/ui/button"
import { retryAccountIdentity } from "@/features/accounts/actions/account-actions"

export function RetryIdentityButton({
  accountId,
  displayName,
}: {
  accountId: string
  displayName: string
}) {
  const [isPending, startTransition] = useTransition()

  function onClick() {
    startTransition(async () => {
      const result = await retryAccountIdentity(accountId)
      if ("success" in result) {
        toast.success("Identity resolved")
      } else if (result.error === "duplicate") {
        toast.error("This account is already connected")
      } else {
        toast.error("Couldn't resolve identity — try again in a minute")
      }
    })
  }

  return (
    <Button
      size="sm"
      onClick={onClick}
      disabled={isPending}
      aria-label={`Retry identity resolution for ${displayName}`}
    >
      {isPending ? (
        <Loader2 className="mr-1 h-3.5 w-3.5 animate-spin" />
      ) : (
        <RefreshCw className="mr-1 h-3.5 w-3.5" />
      )}
      Try again
    </Button>
  )
}
```
