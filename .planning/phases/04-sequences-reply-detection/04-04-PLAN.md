```typescript
import type { SupabaseClient } from "@supabase/supabase-js"

export interface MatchedReply {
  prospectId: string
  prospectHandle: string
  userId: string
  replySnippet: string
}

/**
 * Match a reply sender to a prospect record.
 * Normalizes handles to lowercase for case-insensitive matching.
 * Only matches prospects with pipeline_status != 'replied'.
 */
export async function matchReplyToProspect(
  supabase: SupabaseClient,
  senderHandle: string,
  platform: string,
  accountUserId: string,
): Promise<MatchedReply | null> {
  // Normalize: strip leading "u/" or "U/" if present, lowercase
  const normalized = senderHandle
    .replace(/^u\//i, "")
    .toLowerCase()

  const { data: prospects } = await supabase
    .from("prospects")
    .select("id, handle, user_id, pipeline_status")
    .eq("user_id", accountUserId)
    .eq("platform", platform)
    .neq("pipeline_status", "replied")

  if (!prospects?.length) return null

  // Case-insensitive match
  const match = prospects.find(
    (p) => p.handle?.toLowerCase() === normalized
  )

  if (!match) return null

  return {
    prospectId: match.id,
    prospectHandle: match.handle ?? normalized,
    userId: match.user_id,
    replySnippet: "", // Populated by caller with CU response
  }
}
```
